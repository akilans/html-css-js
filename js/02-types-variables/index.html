Types and Variables!


                // var myName = "Akilan" // Old way not recomended 
                // Use const, let -> es6

                let myName = "Akilan"; // String
                let myAge = 34; // number
                let smoking = false; // boolean
                let myProgramming = [ "Python", "NodeJS", "GoLang" ] // array
                let akilan = {
                    name: "Akilan",
                    age: 34,
                    programming: [ "Python", "NodeJS", "GoLang" ],
                    smoking: false
                } // Object


                // Printing strings, Numbers, Arrays, Boolean, Objects
                console.log( myName );
                console.log( myAge );
                console.log( myProgramming );
                console.log( smoking );
                console.log( myProgramming[ 0 ] );
                console.log( akilan.name );
                console.log( akilan.programming[ 0 ] );
                console.log( akilan.smoking );
