DOM manipulation!

            
                function changeBg() {
                    let body = document.querySelector( "body" );
                    body.style.backgroundColor = "red"
                }
        
                let resetColor = () => {
                    let body = document.querySelector( "body" );
                    body.style.backgroundColor = "white";
                }
        
                // best method
                let method2 = document.querySelector( ".method2" );
                method2.addEventListener( "click", changeBg );
        
                let reset_btn = document.querySelector( ".reset" )
                reset_btn.onclick = () => resetColor();
                reset_btn.onmouseover = () => changeBg();
                reset_btn.onmouseleave = () => resetColor();
