Operators!

            

                let a = 1, b = 2, c = 3;
                let strNum = "100"
                let name = "Akilan";
                let intro = "My name is "
        
                // Addition
                console.groupCollapsed( "Addition" );
                    console.log(a + b); //3
                    console.log(a + strNum); //1100
                    console.log(intro + name); //My name is Akilan
                    console.log(name + " is a DevOps engineer");
                console.groupEnd()
        
                // Subtraction
                console.groupCollapsed( "Subtraction" );
                    console.log(b - a); //1
                    console.log(c -a); //2
                console.groupEnd()
        
                // Multiplication
                console.groupCollapsed( "Multiplication" );
                    console.log(a * b); //2
                    console.log(b * c); //6
                console.groupEnd()
        
                // Incrementing
                console.groupCollapsed( "Incrementing" );
                    console.log(a + 1); //2
                    console.log(a++); //1
                    console.log(a); //2
                    console.log(++b); //3
                    console.log(c--); //3
                    console.log(c); //2
                console.groupEnd()
