Comparison!

            
                let myName = "akilan";
                let newName = "AKILAN";
                let age = 34;
                let strAge = "34";
        
                // String and Number Comparison
                console.groupCollapsed( "String and Number Comparison" );
                    console.log(myName == "akilan") // true
                    console.log(myName == newName) // false
                    console.log(myName == newName.toLowerCase()) // true
                    console.log(age == strAge) // true
                    console.log(age === strAge) // false // check type as well
                    console.log(myName != "akilan") // false
                    console.log(age !== strAge) // true
                console.groupEnd()
        
                let workLocation; // undefined
                let company = null;
        
                // Boolean comparison
                console.groupCollapsed( "Boolean comparison" );
                    console.log(Boolean(workLocation)); //false
                    console.log(Boolean(company)); //false
                    console.log(Boolean(myName)); //true
                    console.log(!Boolean(workLocation)); //true
                console.groupEnd()
        
                let numbers1 = [1,2,3];
                let numbers2 = [1,2,3];
                let rose1 = { color: "red"}
                let rose2 = { color: "red"}
        
                // Object and Array Comparison
                console.groupCollapsed( "Object and Array Comparison" );
                    console.log(numbers1 == numbers2) // false
                    console.log(rose1 == rose2); //false
                console.groupEnd()
        
        
                // AND OR comparison
                console.groupCollapsed( "AND OR comparison" );
                    // AND both must be true
                    console.log(Boolean("") && Boolean("")) // false
                    console.log(Boolean(myName) && Boolean("")) // false
                    console.log(Boolean(myName) && Boolean(age)) // true
        
                    //OR atleast one should be true
                    console.log(Boolean("") || Boolean("")) // false
                    console.log(Boolean(myName) || Boolean("")) // true
                    console.log(Boolean(myName) || Boolean(age)) // true
        
                console.groupEnd()
